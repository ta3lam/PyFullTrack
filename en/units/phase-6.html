import pandas as pd, numpy as np np.random.seed(42) n = 100 df = pd.DataFrame({ "Sales": np.random.normal(50000,10000,n), "Marketing": np.random.normal(5000,1000,n), "Staff": np.random.randint(5,50,n), "Season": np.random.choice([1,2,3,4],n), }) # Add correlation df["Sales"] += df["Marketing"] * 4 corr = df.corr() print(corr.round(2)) # Find top correlated pairs import numpy as np mask = np.triu(np.ones(corr.shape), k=1).astype(bool) pairs = corr.where(mask).stack().sort_values(ascending=False) print("\nTop correlations:") print(pairs.head(3))

Simple Linear Regression

python
import numpy as np # Manual linear regression X = df["Marketing"].values y = df["Sales"].values # OLS: y = a + bX b = np.cov(X, y)[0,1] / np.var(X) a = y.mean() - b * X.mean() # Predictions y_pred = a + b * X residuals = y - y_pred r_squared = 1 - (residuals**2).sum() / ((y - y.mean())**2).sum() print(f"Intercept (a): {a:,.0f}") print(f"Slope (b): {b:.2f}") print(f"R²: {r_squared:.3f}") print(f"\nFor 6000 marketing: {a + b*6000:,.0f} predicted sales")

Rolling Windows

python
import pandas as pd, numpy as np monthly = pd.Series([45,52,48,61,58,72,65,59,71,78,82,95], index=pd.date_range("2024-01",periods=12,freq="ME"), name="Sales" ) * 1000 # Rolling mean (smoothing) ma3 = monthly.rolling(window=3).mean() # 3-month moving avg ma6 = monthly.rolling(window=6).mean() # 6-month moving avg # Rolling statistics roll_std = monthly.rolling(window=3).std() roll_min = monthly.rolling(window=3).min() roll_max = monthly.rolling(window=3).max() result = pd.DataFrame({ "Sales": monthly, "MA3": ma3, "MA6": ma6, "Std3": roll_std, }).round(0) print(result.head(8))

Expanding & Cumulative

# Expanding — includes all previous rows cum_mean = monthly.expanding().mean() cum_total = monthly.cumsum() cum_max = monthly.cummax() # Percentage change pct = monthly.pct_change() * 100 result2 = pd.DataFrame({ "Sales": monthly, "CumTotal": cum_total, "RunningAvg":cum_mean, "Growth%": pct, }).round(0) print(result2.tail(5))
💡 Tip: Rolling averages smooth out noise in time series. Use 3-period MA for short-term trend, 12-period for long-term.

📊 Sampling & Hypothesis Testing

Random Sampling

python
import pandas as pd, numpy as np np.random.seed(42) # Large population population = pd.DataFrame({ "Sales": np.random.normal(50000,15000,10000).round(0), "Region": np.random.choice(["Cairo","Alex","Giza"],10000), }) # Simple random sample sample = population.sample(n=100, random_state=42) print(f"Population mean: {population['Sales'].mean():,.0f}") print(f"Sample mean: {sample['Sales'].mean():,.0f}") # Stratified sample (proportional by region) stratified = population.groupby("Region", group_keys=False).apply( lambda x: x.sample(frac=0.01, random_state=42) ) print(f"\nStratified sample: {len(stratified)} rows") print(stratified["Region"].value_counts())
Output
Population mean: 50,012 Sample mean: 49,847 Stratified sample: 100 rows Region Cairo 34 Alex 33 Giza 33

Hypothesis Testing (t-test)

python
import numpy as np from scipy import stats # Are Cairo sales significantly higher than Alex? cairo = population[population["Region"]=="Cairo"]["Sales"] alex = population[population["Region"]=="Alex"]["Sales"] t_stat, p_value = stats.ttest_ind(cairo, alex) print(f"Cairo mean: {cairo.mean():,.0f}") print(f"Alex mean: {alex.mean():,.0f}") print(f"t-statistic: {t_stat:.3f}") print(f"p-value: {p_value:.4f}") print(f"Significant (p<0.05): {p_value < 0.05}")
💡 Tip: p-value < 0.05 means we reject the null hypothesis — the difference is statistically significant (not due to random chance).
🎯 TASKPractice Exercise
Compare sales between two marketing campaigns using hypothesis testing. Determine if the difference is statistically significant.
Starter code:
python
import numpy as np, pandas as pd from scipy import stats np.random.seed(42) # Two campaigns campaign_A = np.random.normal(52000, 8000, 50) # New campaign campaign_B = np.random.normal(48000, 9000, 50) # Old campaign # Descriptive stats print("Campaign A:") print(f" Mean: {campaign_A.mean():,.0f} Std: {campaign_A.std():,.0f}") print("Campaign B:") print(f" Mean: {campaign_B.mean():,.0f} Std: {campaign_B.std():,.0f}") print(f" Difference: {campaign_A.mean()-campaign_B.mean():,.0f}") # Independent t-test t_stat, p_val = stats.ttest_ind(campaign_A, campaign_B) print(f"\nt-statistic: {t_stat:.3f}") print(f"p-value: {p_val:.4f}") if p_val < 0.05: print("✅ SIGNIFICANT: Campaign A performs better (p < 0.05)") else: print("❌ NOT significant: Could be random variation (p >= 0.05)") # Effect size (Cohen's d) pooled_std = np.sqrt((campaign_A.std()**2 + campaign_B.std()**2) / 2) cohens_d = (campaign_A.mean() - campaign_B.mean()) / pooled_std print(f"Cohen's d: {cohens_d:.3f} ({'large' if abs(cohens_d)>0.8 else 'medium' if abs(cohens_d)>0.5 else 'small'} effect)")
✅ Solution
python
import numpy as np; from scipy import stats np.random.seed(42) A=np.random.normal(52000,8000,50); B=np.random.normal(48000,9000,50) t,p=stats.ttest_ind(A,B) print(f"A mean: {A.mean():,.0f} | B mean: {B.mean():,.0f}") print(f"p-value: {p:.4f} | Significant: {p<0.05}") d=(A.mean()-B.mean())/np.sqrt((A.std()**2+B.std()**2)/2) print(f"Cohen's d: {d:.3f}")
Output
A mean: 52,847 | B mean: 48,614 p-value: 0.0031 | Significant: True Cohen's d: 0.498

Structuring an Analysis Report

python
import pandas as pd, numpy as np np.random.seed(42) def sales_report(df, title="Sales Report"): """Generate a complete sales analysis report.""" sep = "="*50 print(f"\n{sep}") print(f" {title}") print(sep) # 1. Overview print(f"\n📊 OVERVIEW") print(f" Period: {df.index.min().date()} to {df.index.max().date()}") print(f" Total: {df['Sales'].sum():>15,.0f}") print(f" Daily Avg: {df['Sales'].mean():>15,.0f}") print(f" Best Day: {df['Sales'].max():>15,.0f} ({df['Sales'].idxmax().date()})") # 2. Trend monthly = df["Sales"].resample("ME").sum() growth = monthly.pct_change().mean() * 100 print(f"\n📈 TREND") print(f" Monthly growth (avg): {growth:+.1f}%") # 3. Regional breakdown if "Region" in df.columns: print(f"\n🗺️ BY REGION") by_region = df.groupby("Region")["Sales"].agg(["sum","mean"]).round(0) by_region["Share%"] = (by_region["sum"]/by_region["sum"].sum()*100).round(1) print(by_region.sort_values("sum",ascending=False).to_string()) print(f"\n{sep}\n") return monthly
💡 Tip: A good report has: Overview → Trend → Breakdown → Insights → Recommendations. Always answer the 'so what' question.
