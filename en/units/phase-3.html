What is a Series?

A Series is a 1D labeled array — like a single column in a spreadsheet with an index.

python
import pandas as pd # Create Series sales = pd.Series([45000,52000,48000,61000], index=["Q1","Q2","Q3","Q4"], name="Sales_2024") print(sales) print("\nDtype:", sales.dtype) print("Index:", sales.index.tolist())
Output
Q1 45000 Q2 52000 Q3 48000 Q4 61000 Name: Sales_2024, dtype: int64 Dtype: int64 Index: ['Q1', 'Q2', 'Q3', 'Q4']

Series Operations

python
s = pd.Series({"Cairo":450000,"Alex":280000,"Giza":195000,"Aswan":90000}) print(s["Cairo"]) # 450000 print(s[s > 200000]) # Filter print(s.sort_values()) # Ascending print(s.sort_values(ascending=False).head(2)) # Top 2 # Stats print(f"Total: {s.sum():,}") print(f"Mean: {s.mean():,.0f}") print(f"Max: {s.max():,} ({s.idxmax()})")
Output
450000 Cairo 450000 Alex 280000 dtype: int64 Aswan 90000 Giza 195000 Alex 280000 Cairo 450000 dtype: int64 Cairo 450000 Alex 280000 dtype: int64 Total: 1,015,000 Mean: 253,750 Max: 450,000 (Cairo)

loc and iloc

python
# loc — label-based (use column NAMES) df.loc[0] # Row with index 0 df.loc[0, "Product"] # Specific cell df.loc[0:5, ["Product","Sales"]] # Range + columns # iloc — position-based (use NUMBERS) df.iloc[0] # First row df.iloc[0, 2] # Row 0, column 2 df.iloc[:5, :3] # First 5 rows, first 3 cols df.iloc[[0,2,4], :] # Specific rows # query() — SQL-like syntax result = df.query("Sales > 50000 and Region == 'Cairo'") print(result.head())

Date Index & Resampling

python
import pandas as pd, numpy as np np.random.seed(42) # Create daily time series dates = pd.date_range("2024-01-01","2024-12-31",freq="D") df = pd.DataFrame({ "Sales": np.random.normal(50000,10000,len(dates)).round(0), "Units": np.random.randint(1,20,len(dates)), }, index=dates) # Extract date components df["Month"] = df.index.month df["Quarter"] = df.index.quarter df["DayName"] = df.index.day_name() # Resample — aggregate by different frequencies monthly = df["Sales"].resample("ME").sum() weekly = df["Sales"].resample("W").mean() quarterly = df["Sales"].resample("QE").agg(["sum","mean","count"]) print("Monthly Sales (first 3):") print(monthly.head(3).round(0)) print("\nQuarterly:") print(quarterly.round(0))

import pandas as pd, numpy as np np.random.seed(42) dates = pd.date_range("2023-01-01","2024-12-31",freq="D") # Add trend + seasonality + noise trend = np.linspace(40000,60000,len(dates)) seasonal = np.sin(np.linspace(0,4*np.pi,len(dates))) * 5000 noise = np.random.normal(0,3000,len(dates)) df = pd.DataFrame({"Sales": (trend+seasonal+noise).round(0)}, index=dates) # Monthly totals and growth monthly = df["Sales"].resample("ME").sum() growth = monthly.pct_change() * 100 # Day of week analysis df["DayName"] = df.index.day_name() by_day = df.groupby("DayName")["Sales"].mean().sort_values(ascending=False) # Best and worst months print(f"Best month: {monthly.idxmax().strftime('%Y-%m')} ({monthly.max():,.0f})") print(f"Worst month: {monthly.idxmin().strftime('%Y-%m')} ({monthly.min():,.0f})") print(f"\nAvg growth: {growth.mean():.2f}% per month") print(f"\nBest day of week: {by_day.index[0]} ({by_day.iloc[0]:,.0f})") print(f"Worst day: {by_day.index[-1]} ({by_day.iloc[-1]:,.0f})")
